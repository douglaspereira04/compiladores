PROGRAM -> { COMANDOS }
COMANDOS -> COMANDO ; COMANDOS
COMANDOS -> &
COMANDO -> VAR
COMANDO -> IF
COMANDO -> WHILE
COMANDO -> ATR
VAR -> type id
ATR -> id atrop EXPR
EXPR -> EXPRVAL' EXPR'
EXPR' -> op EXPRVAL' EXPR'
EXPR' -> &
EXPRVAL' -> EXPRVAL
EXPRVAL -> ( EXPR )
EXPRVAL -> id
EXPRVAL -> num
EXPRVAL -> boolean
RELEXPR -> EXPR relop EXPR
IF -> if ( RELEXPR ) { COMANDOS }
WHILE -> while ( RELEXPR ) { COMANDOS }
DOWHILE -> do { COMANDOS } while ( RELEXPR )
